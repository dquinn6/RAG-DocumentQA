 57

 77    def load_csv_file(
 78        self, file_path: str, shuffle: bool = True, seed: Optional[int] = None
 79    ) -> Document:
 80        """Load the CSV data using Langchain's loader.
 81
 82        Args:
 83            file_path (str): Path to processed CSV data.
 84            shuffle (bool, optional): Shuffle the data. Defaults to True.
 85            seed (Optional[int, optional): Random seed for shuffle. Defaults to None.
 86
 87        Returns:
 88            Document: Data from CSV loader.
 89        """
 90        # keeping this function within this class since it uses the langchain loader
 91        try:
 92            loader = CSVLoader(
 93                file_path=file_path,
 94                encoding="utf-8",
 95                csv_args={"delimiter": ","},
 96            )
 97            csv_data = loader.load()
 98            if shuffle:
 99                random.seed(seed)
100                random.shuffle(csv_data)
101
102        except Exception as e:
103            raise VectorstoreError(f"Failed to load csv data: {e}")
104
105        return csv_data

107    def chunk_data(
108        self, chunk_size: int = 2048, chunk_overlap: int = 50
109    ) -> None:
110        """Chunk the data to specified chunk size using RecursiveCharacterTextSplitter.
111
112        Args:
113            chunk_size (int, optional): Token size to chunk data to. Defaults to 2048.
114            chunk_overlap (int, optional): Token overlap between chunks. Defaults to 50.
115        """
116        try:
117            text_splitter = RecursiveCharacterTextSplitter(
118                chunk_size=chunk_size,
119                chunk_overlap=chunk_overlap,
120            )
121            self.data = text_splitter.split_documents(self.data)
122            if self.verbose:
123                logging.info(f"Data chunked to size {chunk_size}")
124
125        except Exception as e:
126            logging.error(f"Failed to chunk data: {e}")

128    def create_local_vectorstore(
129        self,
130        save_path: str,
131        force_create: bool = False,
132        callback: Optional[Callable] = None,
133    ) -> None:
134        """Create a new local vectorstore index.
135
136        Args:
137            save_path (str): Path to save vectorstore index to.
138            force_create (bool, optional): Forces creation of vectostore and avoid 'already found; load instead?' check. Defaults to False.
139            callback (Optional[Callable], optional): Optional callback function. Defaults to None.
140
141        Raises:
142            ValueError: Raised when given invalid input for 'already found; load instead?' check.
143            VectorstoreError: Raised when vectorstore creation was unsuccessful.
144        """
145        if os.path.exists(save_path) and not force_create:
146            overwrite_saved = input(
147                f"Vectorstore already found at {save_path}; overwrite? [y/n]: "
148            ).lower()
149            if overwrite_saved not in ["y", "n"]:
150                raise ValueError(
151                    "Invalid input; try again with 'y' for yes or 'n' for no."
152                )
153
154            if overwrite_saved == "n":
155                load_instead = input("Load instead [y/n]: ").lower()
156                if load_instead not in ["y", "n"]:
157                    raise ValueError(
158                        "Invalid input; try again with 'y' for yes or 'n' for no."
159                    )
160
161                if load_instead == "y":
162                    self.load_local_vectorstore(load_path=save_path)
163                else:
164                    if self.verbose:
165                        logging.info(
166                            "Vectorstore not overwritten; aborting... "
167                        )
168
169        else:
170            logging.info(f"Creating a new local vectorstore at: {save_path}")
171            try:
172                # No built in progress bar from their API; using this workaround shared at: https://stackoverflow.com/questions/77836174/how-can-i-add-a-progress-bar-status-when-creating-a-vector-store-with-langchain
173                total_len = len(self.data)
174
175                # Below function is equivalent to embedding each piece of text, zipping text and embeddings as pairs, and creating index from these pairs
176                with tqdm(
177                    total=total_len, desc="Processing documents"
178                ) as progress_bar:
179                    for d in range(total_len):
180                        doc = self.data[d]
181                        # If already init, add next doc
182                        if self.vectorstore:
183                            self.vectorstore.add_documents([doc])
184                        else:  # Init vectorstore with first doc
185                            self.vectorstore = FAISS.from_documents(
186                                [doc], self.embedding_type
187                            )
188
189                        # Update progress bar and optional callback for higher level (UI) progress bars.
190                        progress_bar.update(1)
191                        if callback:
192                            callback(d / total_len, d)
193
194                self.vectorstore.save_local(save_path)
195                if self.verbose:
196                    logging.info(
197                        f"Vectorstore successfully set and saved to {save_path}"
198                    )
199
200            except Exception as e:
201                raise VectorstoreError(f"Failed to create vectorstore: {e}")
202
203    def load_local_vectorstore(self, load_path: str) -> None:
204        """Load previously generated local vectorstore index.
205
206        Args:
207            load_path (str): Path to local vectorstore.
208
209        Raises:
210            FileNotFoundError: Raised when index cannot be found at load_path.
211            VectorstoreError: Raised when vectorstore load was unsuccessful.
212        """
213        if not os.path.exists(load_path):
214            raise FileNotFoundError(
215                f"Failed to find a saved vectorstore at {load_path}; please ensure save_path points to correct location."
216            )
217
218        try:
219            self.vectorstore = FAISS.load_local(
220                load_path,
221                self.embedding_type,
222                allow_dangerous_deserialization=True,
223            )
224            if self.verbose:
225                logging.info("Vectorstore loaded successfully.")
226
227        except Exception as e:
228            raise VectorstoreError(f"Failed to load vectorstore: {e}")

230    def create_retriever(
231        self,
232        search_type: str = "similarity",
233        search_kwargs: Optional[dict] = None,
234    ) -> None:
235        """Create document retriever object for vectorstore.
236
237        Args:
238            search_type (str, optional): Algorithm to use when searching for related documents. Defaults to "similarity".
239            search_kwargs (Optional[dict], optional): Retriever kwargs, which can include 'k' for number of retrieved docs. Defaults to None.
240
241        Raises:
242            VectorstoreError: Raised when vectorstore was initialized but never loaded/created.
243            ValueError: Raised when given an invalid arg for search_type.
244            VectorstoreError: Raised when retriever creation was unsuccessful.
245        """
246        if self.vectorstore is None:
247            raise VectorstoreError(
248                "Vectorstore not set; create or load a vectorstore using class method first."
249            )
250
251        search_types = ["similarity", "mmr", "similarity_score_threshold"]
252        if search_type not in search_types:
253            raise ValueError(
254                f"Invalid arg for search_type; valid args include: {search_types}"
255            )
256
257        try:
258            self.retriever = self.vectorstore.as_retriever(
259                search_type=search_type,
260                search_kwargs=search_kwargs,
261            )
262            if self.verbose:
263                logging.info("Retriever successfully set")
264
265        except Exception as e:
266            raise VectorstoreError(f"Failed to create retriever: {e}")
267            # logging.error(f"Failed to create retriever: {e}")

269    def retrieve_top_documents(self, query: str) -> List[str]:
270        """Use retriever to fetch top k matching documents for RAG.
271
272        Args:
273            query (str): Query to search documents against.
274
275        Raises:
276            VectorstoreError: Raised when retriever was never created.
277
278        Returns:
279            Optional[List[str]]: List of retrieved documents for RAG.
280        """
281        if self.retriever is None:
282            raise VectorstoreError(
283                "Retriver not set; create a retriever using class method first"
284            )
285
286        try:
287            retrieved_docs = self.retriever.get_relevant_documents(query)
288        except Exception as e:
289            raise VectorstoreError(f"Failed to retrive documents: {e}")
290
291        return [
292            retrieved_docs[i].page_content for i in range(len(retrieved_docs))
293        ]

137        Args:

145        """
146        if os.path.exists(save_path) and not force_create:
147            overwrite_saved = input(
148                f"Vectorstore already found at {save_path}; overwrite? [y/n]: "
149            ).lower()
150            if overwrite_saved not in ["y", "n"]:
151                raise ValueError(
152                    "Invalid input; try again with 'y' for yes or 'n' for no."
153                )

155            if overwrite_saved == "n":

160                    )
161

163                    self.load_local_vectorstore(load_path=save_path)
164                else:

166                        logging.info(
167                            "Vectorstore not overwritten; aborting... "
168                        )

171            logging.info(f"Creating a new local vectorstore at: {save_path}")
172            try:
173                # No built in progress bar from their API; using this workaround shared at: https://stackoverflow.com/questions/77836174/how-can-i-add-a-progress-bar-status-when-creating-a-vector-store-with-langchain
174                total_len = len(self.data)
175
176                # Below function is equivalent to embedding each piece of text, zipping text and embeddings as pairs, and creating index from these pairs
177                with tqdm(
178                    total=total_len, desc="Processing documents"
179                ) as progress_bar:
180                    for d in range(total_len):
181                        doc = self.data[d]
182                        # If already init, add next doc
183                        if self.vectorstore:
184                            self.vectorstore.add_documents([doc])
185                        else:  # Init vectorstore with first doc
186                            self.vectorstore = FAISS.from_documents(
187                                [doc], self.embedding_type
188                            )
189
190                        # Update progress bar and optional callback for higher level (UI) progress bars.
191                        progress_bar.update(1)
192                        if callback:
193                            callback(d / total_len, d)
194
195                self.vectorstore.save_local(save_path)
196                if self.verbose:
197                    logging.info(
198                        f"Vectorstore successfully set and saved to {save_path}"
199                    )
200
201            except Exception as e:
